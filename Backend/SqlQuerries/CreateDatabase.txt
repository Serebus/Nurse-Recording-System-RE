-- ==========================================
-- 1. Create Database
-- ==========================================
CREATE DATABASE NurseRecordingSystem;
GO

USE NurseRecordingSystem;
GO

-- ==========================================
-- 2. Create Tables
-- ==========================================

-- Auth Table (Handles credentials)
CREATE TABLE [Auth] (
    AuthId INT IDENTITY(1,1) PRIMARY KEY,
    UserName NVARCHAR(100) NOT NULL,
    PasswordHash VARBINARY(MAX) NOT NULL,
    PasswordSalt VARBINARY(MAX) NOT NULL,
    Email NVARCHAR(255) NOT NULL UNIQUE,
    CreatedOn DATETIME DEFAULT GETUTCDATE(),
    CreatedBy NVARCHAR(100),
    UpdatedOn DATETIME,
    UpdatedBy NVARCHAR(100),
    IsActive BIT DEFAULT 1
);
GO

-- Users Table (Handles personal information)
CREATE TABLE [Users] (
    UserId INT IDENTITY(1,1) PRIMARY KEY,
    AuthId INT NOT NULL FOREIGN KEY REFERENCES [Auth](AuthId),
    FirstName NVARCHAR(100) NOT NULL,
    MiddleName NVARCHAR(100),
    LastName NVARCHAR(100) NOT NULL,
    ContactNumber NVARCHAR(20),
    Address NVARCHAR(MAX),
    Role NVARCHAR(50) DEFAULT 'User',
    CreatedOn DATETIME DEFAULT GETUTCDATE(),
    CreatedBy NVARCHAR(100),
    UpdatedOn DATETIME,
    UpdatedBy NVARCHAR(100),
    IsActive BIT DEFAULT 1
);
GO

-- Session Token Table
CREATE TABLE [SessionToken] (
    TokenId INT IDENTITY(1,1) PRIMARY KEY,
    Token VARBINARY(64) NOT NULL UNIQUE,
    AuthId INT NOT NULL FOREIGN KEY REFERENCES [Auth](AuthId),
    ExpiresOn DATETIME NOT NULL,
    IsActive BIT DEFAULT 1
);
GO

-- Patient Form Table
CREATE TABLE [PatientForm] (
    FormId INT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NOT NULL FOREIGN KEY REFERENCES [Users](UserId),
    PatientName NVARCHAR(200) NOT NULL,
    IssueType NVARCHAR(100) NOT NULL,
    -- Spelled exactly as expected by the C# GetOrdinal("issueDescryption") mapping
    IssueDescryption NVARCHAR(MAX), 
    Status NVARCHAR(50) DEFAULT 'Pending',
    CreatedOn DATETIME DEFAULT GETUTCDATE(),
    CreatedBy NVARCHAR(100),
    UpdatedOn DATETIME,
    UpdatedBy NVARCHAR(100),
    IsActive BIT DEFAULT 1
);
GO

-- Clinic Status Table
CREATE TABLE [dbo].[ClinicStatus] (
    [LogId] INT IDENTITY(1,1) PRIMARY KEY,
    [Status] BIT NOT NULL,
    [CreatedOn] DATETIME DEFAULT GETDATE() 
);
GO
