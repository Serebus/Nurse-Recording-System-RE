IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'hsp_ViewUserForm')
    DROP PROCEDURE dbo.hsp_ViewUserForm;
GO

CREATE PROCEDURE dbo.hsp_ViewUserForm
    @formId INT
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT EXISTS (SELECT 1 FROM [PatientForm] WHERE FormId = @formId AND IsActive = 1)
    BEGIN
        -- Error matched by C# exception handling checking error code 50001
        RAISERROR('Patient form not found or is archived.', 16, 1);
        RETURN;
    END

    SELECT
        FormId AS formId,
        IssueType AS issueType,
        IssueDescryption AS issueDescryption,
        Status AS status,
        UserId AS userId,
        PatientName AS patientName,
        CreatedOn AS createdOn,
        CreatedBy AS createdBy,
        UpdatedOn AS updatedOn,
        UpdatedBy AS updatedBy,
        IsActive AS isActive
    FROM [PatientForm]
    WHERE FormId = @formId AND IsActive = 1;
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'nsp_ViewAllUsers')
    DROP PROCEDURE dbo.nsp_ViewAllUsers;
GO

CREATE PROCEDURE dbo.nsp_ViewAllUsers
    @IsActive BIT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        u.UserId AS userId,
        a.UserName AS userName,
        a.Email AS email,
        u.Role AS role,
        u.FirstName AS firstName,
        u.MiddleName AS middleName,
        u.LastName AS lastName,
        u.ContactNumber AS contactNumber,
        u.Address AS address,
        u.IsActive AS isActive,
        u.CreatedOn AS createdOn,
        u.CreatedBy AS createdBy,
        u.UpdatedOn AS updatedOn,
        u.UpdatedBy AS updatedBy
    FROM [Users] u
    INNER JOIN [Auth] a ON u.AuthId = a.AuthId
    WHERE (@IsActive IS NULL OR u.IsActive = @IsActive);
END;
GO


CREATE PROCEDURE [dbo].[hsp_ViewClinicStatus]
AS
BEGIN
    SET NOCOUNT ON;

    -- Replace [ClinicStatus] with your actual table name
    -- if it is different in your database.
    SELECT 
        LogId, 
        Status 
    FROM 
        [dbo].[ClinicStatus]
    ORDER BY 
        LogId DESC; -- Optional: orders by newest first
END;
GO

IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'hsp_ViewAllMedecineStock')
    DROP PROCEDURE dbo.hsp_ViewAllMedecineStock;
GO

CREATE PROCEDURE [dbo].[hsp_ViewAllMedecineStock]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT 
        StockId AS medicineId,
        MedicineName AS medicineName,
        Description AS medecineDescription,
        Quantity AS numberOfStock,
        ExpirationDate AS expirationDate,
        CreatedOn AS createdOn,
        CreatedBy AS createdBy,
        UpdatedOn AS updatedOn,
        UpdatedBy AS updatedBy,
        IsActive AS isActive
    FROM [dbo].[MedecineStock];
END;
GO


CREATE PROCEDURE [dbo].[asp_CreateClinicStatus]
    @status BIT
AS
BEGIN
    SET NOCOUNT ON;

    -- Insert the new clinic status
    INSERT INTO [dbo].[ClinicStatus] ([Status])
    VALUES (@status);

    -- Return the newly created LogId back to the C# application
    SELECT SCOPE_IDENTITY() AS LogId;
END;
GO
