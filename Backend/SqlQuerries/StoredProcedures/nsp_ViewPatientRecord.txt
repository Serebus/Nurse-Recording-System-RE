CREATE PROCEDURE [dbo].[nsp_ViewPatientRecord]
    @patientRecordId INT
AS
BEGIN
    SET NOCOUNT ON;

    SELECT 
        [patientRecordId] AS PatientRecordId,
        [nursingDiagnosis] AS NursingDiagnosis,
        [nursingIntervention] AS NursingIntervention,
        [nurseId] AS NurseId,
        [createdOn] AS CreatedOn,
        [createdBy] AS CreatedBy,
        [updatedOn] AS UpdatedOn,
        [updatedBy] AS UpdatedBy,
        [isActive] AS IsActive
    FROM [PatientRecords]
    WHERE [patientRecordId] = @patientRecordId AND [isActive] = 1;
END
GO
