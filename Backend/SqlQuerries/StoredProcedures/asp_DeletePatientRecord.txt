CREATE PROCEDURE [dbo].[asp_DeletePatientRecord]
    @patientRecordId INT,
    @deletedBy NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE [PatientRecords]
    SET [isActive] = 0,
        [updatedOn] = SYSUTCDATETIME(),
        [updatedBy] = @deletedBy
    WHERE [patientRecordId] = @patientRecordId AND [isActive] = 1;
END
GO
