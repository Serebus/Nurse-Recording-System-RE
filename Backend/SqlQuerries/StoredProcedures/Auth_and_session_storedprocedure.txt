IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'usp_CreateUserAndAuth')
    DROP PROCEDURE dbo.usp_CreateUserAndAuth;
GO

CREATE PROCEDURE dbo.usp_CreateUserAndAuth
    @passwordHash VARBINARY(MAX),
    @passwordSalt VARBINARY(MAX),
    @email NVARCHAR(255),
    @createdBy NVARCHAR(50),
    @updatedOn DATETIME,
    @updatedBy NVARCHAR(50),
    @isActive BIT,
    @firstName NVARCHAR(50),
    @lastName NVARCHAR(50),
    @emergencyContact NVARCHAR(20),
    @address NVARCHAR(255),
    @facebook NVARCHAR(255) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @AuthId INT;

    -- First, check if the email already exists to provide a clean error.
    IF EXISTS (SELECT 1 FROM [Auth] WHERE Email = @email)
    BEGIN
        RAISERROR('Email already existing', 16, 1);
        RETURN -1; -- Indicates failure
    END

    BEGIN TRY
        BEGIN TRANSACTION;
        -- 1. Insert into the Auth table
        -- We use the email for the UserName column to satisfy the schema, as it's no longer sent from the app.
        INSERT INTO dbo.Auth (
            UserName,
            PasswordHash,
            PasswordSalt,
            Email,
            CreatedBy,
            CreatedOn,
            UpdatedBy,
            UpdatedOn,
            IsActive
        )
        VALUES (
            @email, -- Using email for UserName
            @passwordHash,
            @passwordSalt,
            @email,
            @createdBy,
            GETUTCDATE(), -- Use the server's UTC time for consistency
            @updatedBy,
            @updatedOn,
            @isActive
        );

        SET @AuthId = SCOPE_IDENTITY();

        -- 2. Insert into the Users table using the updated schema columns
        INSERT INTO dbo.Users (AuthId, FirstName, MiddleName, LastName, EmergencyContact, Address, Facebook, Role)
        VALUES (@AuthId, @firstName, NULL, @lastName, @emergencyContact, @address, @facebook, 'User');

        COMMIT TRANSACTION;

        -- Return the new AuthId on success, which the C# service expects
        SELECT @AuthId;
    END TRY
    BEGIN CATCH
        IF @@TRANCOUNT > 0
            ROLLBACK TRANSACTION;
        ;THROW; -- Re-throw the original error so the C# application can catch it
    END CATCH
END;
GO

    DROP PROCEDURE dbo.usp_LoginUserAuth;

CREATE PROCEDURE dbo.usp_LoginUserAuth
    @email NVARCHAR(255)
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        a.AuthId AS authId,
        a.UserName AS userName,
        a.Email AS email,
        a.PasswordHash AS passwordHash,
        a.PasswordSalt AS passwordSalt,
        u.Role AS role,
        u.UserId AS userId,
        u.FirstName AS User_firstName,
        u.MiddleName AS User_middleName,
        u.LastName AS User_lastName,
        u.EmergencyContact AS User_emergencyContact,
        u.Address AS User_address,
        u.Facebook AS User_facebook,
        CAST(NULL AS INT) AS nurseId,
        CAST(NULL AS NVARCHAR(100)) AS Nurse_firstName,
        CAST(NULL AS NVARCHAR(100)) AS Nurse_middleName,
        CAST(NULL AS NVARCHAR(100)) AS Nurse_lastName,
        CAST(NULL AS NVARCHAR(20)) AS Nurse_contactNumber
    FROM [Auth] a
    LEFT JOIN [Users] u ON a.AuthId = u.AuthId
    WHERE a.Email = @email
      AND a.IsActive = 1;
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'usp_CreateSessionToken')
    DROP PROCEDURE dbo.usp_CreateSessionToken;
GO

CREATE PROCEDURE dbo.usp_CreateSessionToken
    @authId INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Deactivate any previously active tokens for this user
    UPDATE [SessionToken] SET IsActive = 0 WHERE AuthId = @authId AND IsActive = 1;

    DECLARE @NewToken VARBINARY(64) = CRYPT_GEN_RANDOM(64);
    DECLARE @ExpiresOn DATETIME = DATEADD(HOUR, 2, GETUTCDATE()); -- 2 Hours Expiration

    INSERT INTO [SessionToken] (Token, AuthId, ExpiresOn, IsActive)
    VALUES (@NewToken, @authId, @ExpiresOn, 1);

    SELECT
        TokenId AS tokenId,
        Token AS token,
        AuthId AS authId,
        ExpiresOn AS expiresOn,
        CAST(IsActive AS INT) AS isActive -- C# maps this to int
    FROM [SessionToken]
    WHERE TokenId = SCOPE_IDENTITY();
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'usp_UpdateSessionToken')
    DROP PROCEDURE dbo.usp_UpdateSessionToken;
GO

CREATE PROCEDURE dbo.usp_UpdateSessionToken
    @authId INT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @NewToken VARBINARY(64) = CRYPT_GEN_RANDOM(64);
    DECLARE @ExpiresOn DATETIME = DATEADD(HOUR, 2, GETUTCDATE());

    UPDATE [SessionToken]
    SET Token = @NewToken, ExpiresOn = @ExpiresOn
    WHERE AuthId = @authId AND IsActive = 1 AND ExpiresOn > GETUTCDATE();

    SELECT
        TokenId AS tokenId,
        Token AS token,
        AuthId AS authId,
        ExpiresOn AS expiresOn,
        CAST(IsActive AS INT) AS isActive
    FROM [SessionToken]
    WHERE AuthId = @authId AND IsActive = 1 AND Token = @NewToken;
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'usp_EndSessionToken')
    DROP PROCEDURE dbo.usp_EndSessionToken;
GO

CREATE PROCEDURE dbo.usp_EndSessionToken
    @Token VARBINARY(64)
AS
BEGIN
    SET NOCOUNT ON;
    UPDATE [SessionToken]
    SET IsActive = 0
    WHERE Token = @Token AND IsActive = 1;
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'usp_ValidateSessionToken')
    DROP PROCEDURE dbo.usp_ValidateSessionToken;
GO

CREATE PROCEDURE dbo.usp_ValidateSessionToken
    @authId INT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @IsValid BIT = 0;

    IF EXISTS (SELECT 1 FROM [SessionToken] WHERE AuthId = @authId AND IsActive = 1 AND ExpiresOn > GETUTCDATE())
    BEGIN
        SET @IsValid = 1;
    END

    SELECT @IsValid;
END;
GO


IF EXISTS (SELECT 1 FROM sys.procedures WHERE name = 'sp_ValidateSessionTokenAndGetRole')
    DROP PROCEDURE dbo.sp_ValidateSessionTokenAndGetRole;
GO

CREATE PROCEDURE dbo.sp_ValidateSessionTokenAndGetRole
    @Token VARBINARY(64)
AS
BEGIN
    SET NOCOUNT ON;

    SELECT u.Role
    FROM [SessionToken] st
    INNER JOIN [Auth] a ON st.AuthId = a.AuthId
    INNER JOIN [Users] u ON a.AuthId = u.AuthId
    WHERE st.Token = @Token AND st.IsActive = 1 AND st.ExpiresOn > GETUTCDATE();
END;
GO

CREATE PROCEDURE dbo.asp_CreateNurse
    @userName NVARCHAR(100),
    @passwordHash VARBINARY(MAX),
    @passwordSalt VARBINARY(MAX),
    @email NVARCHAR(255),
    @firstName NVARCHAR(100),
    @middleName NVARCHAR(100) = NULL,
    @lastName NVARCHAR(100),
    @contactNumber NVARCHAR(20) = NULL,
    @address NVARCHAR(MAX) = NULL,
    @createdBy NVARCHAR(100) = 'System'
AS
BEGIN
    SET NOCOUNT ON;

    -- 1. Check if the email is already registered 
    -- (Throws error 50005 so your C# backend catches it as an InvalidOperationException)
    IF EXISTS (SELECT 1 FROM [Auth] WHERE Email = @email)
    BEGIN
        ;THROW 50005, 'Email already exists.', 1;
    END

    BEGIN TRY
        BEGIN TRANSACTION;

        DECLARE @NewAuthId INT;

        -- 2. Insert into the Auth Table
        INSERT INTO [Auth] (UserName, PasswordHash, PasswordSalt, Email, CreatedBy, CreatedOn, IsActive)
        VALUES (@userName, @passwordHash, @passwordSalt, @email, @createdBy, GETUTCDATE(), 1);

        SET @NewAuthId = SCOPE_IDENTITY();

        -- 3. Insert into the Users Table with the Role explicitly set to 'Nurse'
        INSERT INTO [Users] (AuthId, FirstName, MiddleName, LastName, EmergencyContact, Address, Role, CreatedBy, CreatedOn, IsActive)
        VALUES (@NewAuthId, @firstName, @middleName, @lastName, @contactNumber, @address, 'Nurse', @createdBy, GETUTCDATE(), 1);

        COMMIT TRANSACTION;

        -- 4. Return the newly created AuthId back to the C# service
        SELECT @NewAuthId AS AuthId;

    END TRY
    BEGIN CATCH
        -- Rollback the transaction if anything fails (e.g., truncation, missing fields)
        IF @@TRANCOUNT > 0
            ROLLBACK TRANSACTION;
        
        -- Rethrow the actual SQL error so the C# API can log it
        ;THROW;
    END CATCH
END;
GO
