CREATE PROCEDURE [dbo].[usp_UpdateFollowUp]
    @FollowUpId INT,
    @FollowUpDate DATETIME2,
    @Description NVARCHAR(MAX),
    @UpdatedBy NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    
    UPDATE [PatientFollowUps]
    SET [followUpDate] = @FollowUpDate,
        [description] = @Description,
        [updatedOn] = SYSUTCDATETIME(),
        [updatedBy] = @UpdatedBy
    WHERE [followUpId] = @FollowUpId AND [isActive] = 1;

    IF @@ROWCOUNT = 0 
        THROW 50001, 'Follow-up not found or inactive.', 1;
END
GO
